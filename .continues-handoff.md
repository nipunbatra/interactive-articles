function eigenDecomposition(sigmaX, sigmaY, rho) {
  const a = sigmaX ** 2;
  const d = sigmaY ** 2;
  const b = rho * sigmaX * sigmaY;
  const trace = a + d;
  const detPart = Math.sqrt((a - d) ** 2 + 4 * b * b);
  const lambda1 = (trace + detPart) / 2;
  const lambda2 = (trace - detPart) / 2;
  const angle = 0.5 * Math.atan2(2 * b, a - d);
  return { lambda1, lambda2, angle };
}

function samplePoints(state) {
  const sqrtTerm = Math.sqrt(Math.max(1 - state.rho ** 2, 0.0001));
  return baseSamples.map(([z1, z2]) => ({
    x: state.muX + state.sigmaX * z1,
    y: state.muY + state.sigmaY * (state.rho * z1 + sqrtTerm * z2),
  }));
}

function getViewBounds(state) {
  const spread = Math.max(state.sigmaX, state.sigmaY) * 4.2;
  return {
    xMin: state.muX - spread,
    xMax: state.muX + spread,
    yMin: state.muY - spread,
    yMax: state.muY + spread,
  };
}

function drawPlane(ctx, width, height, bounds, options = {}) {
  const padding = options.padding ?? 26;
  const paddingLeft = options.paddingLeft ?? padding + 14;
  const paddingRight = options.paddingRight ?? padding;
  const paddingTop = options.paddingTop ?? padding;
  const paddingBottom = options.paddingBottom ?? padding + 14;
  const plotWidth = width - paddingLeft - paddingRight;
  const plotHeight = height - paddingTop - paddingBottom;
  const xToPx = (x) => paddingLeft + ((x - bounds.xMin) / (bounds.xMax - bounds.xMin)) * plotWidth;
  const yToPx = (y) => paddingTop + plotHeight - ((y - bounds.yMin) / (bounds.yMax - bounds.yMin)) * plotHeight;
  const pxToX = (px) => bounds.xMin + ((px - paddingLeft) / plotWidth) * (bounds.xMax - bounds.xMin);
  const pxToY = (py) => bounds.yMin + ((plotHeight - (py - paddingTop)) / plotHeight) * (bounds.yMax - bounds.yMin);

  ctx.fillStyle = 'rgba(255, 255, 255, 0.68)';
  ctx.fillRect(paddingLeft, paddingTop, plotWidth, plotHeight);

  const step = niceStep((bounds.xMax - bounds.xMin) / 6);
  ctx.strokeStyle = 'rgba(30, 36, 48, 0.08)';
  ctx.lineWidth = 1;
  for (let x = Math.ceil(bounds.xMin / step) * step; x <= bounds.xMax + 1e-6; x += step) {
    const px = xToPx(x);
    ctx.beginPath();
    ctx.moveTo(px, paddingTop);
    ctx.lineTo(px, paddingTop + plotHeight);
    ctx.stroke();
  }
  for (let y = Math.ceil(bounds.yMin / step) * step; y <= bounds.yMax + 1e-6; y += step) {
    const py = yToPx(y);
    ctx.beginPath();
    ctx.moveTo(paddingLeft, py);
    ctx.lineTo(paddingLeft + plotWidth, py);
    ctx.stroke();
  }

  ctx.strokeStyle = 'rgba(30, 36, 48, 0.28)';
  ctx.strokeRect(paddingLeft, paddingTop, plotWidth, plotHeight);

  ctx.font = '12px "Avenir Next", sans-serif';
  ctx.fillStyle = 'rgba(94, 102, 118, 0.86)';
  ctx.fillText('x', paddingLeft + plotWidth - 10, paddingTop + plotHeight + 18);
  ctx.fillText('y', paddingLeft - 12, paddingTop + 10);

  return { ctx, width, height, bounds, paddingLeft, paddingRight, paddingTop, paddingBottom, plotWidth, plotHeight, xToPx, yToPx, pxToX, pxToY };
}

function drawGaussianCloud(ctx, plane, state, options = {}) {
  const { xToPx, yToPx } = plane;
  const { lambda1, lambda2, angle } = eigenDecomposition(state.sigmaX, state.sigmaY, state.rho);
  const levels = [3.1, 2.45, 1.8, 1.15];

  ctx.save();
  ctx.translate(xToPx(state.muX), yToPx(state.muY));
  ctx.rotate(-angle);
  levels.forEach((level, index) => {
    const rx = level * Math.sqrt(lambda1) * plane.plotWidth / (plane.bounds.xMax - plane.bounds.xMin);
    const ry = level * Math.sqrt(lambda2) * plane.plotHeight / (plane.bounds.yMax - plane.bounds.yMin);
    ctx.beginPath();
    ctx.ellipse(0, 0, rx, ry, 0, 0, TAU);
    ctx.fillStyle = `rgba(45, 139, 136, ${0.06 + index * 0.04})`;
    ctx.fill();
    ctx.strokeStyle = `rgba(45, 139, 136, ${0.18 + index * 0.1})`;
    ctx.lineWidth = 1.4;
    ctx.stroke();
  });
  ctx.restore();

  const points = samplePoints(state);
  ctx.fillStyle = 'rgba(216, 112, 79, 0.34)';
  points.forEach((point) => {
    const px = xToPx(point.x);
    const py = yToPx(point.y);
    ctx.beginPath();
    ctx.arc(px, py, 2.1, 0, TAU);
    ctx.fill();
  });

  ctx.fillStyle = '#1e2430';
  ctx.beginPath();
  ctx.arc(xToPx(state.muX), yToPx(state.muY), 4.2, 0, TAU);
  ctx.fill();

  if (options.showConditional) {
    const conditionalInfo = getConditionalInfo(state);
    ctx.strokeStyle = 'rgba(216, 112, 79, 0.92)';
    ctx.lineWidth = 2;
    if (state.conditionAxis === 'x') {
      const x = xToPx(conditionalInfo.observedValue);
      ctx.beginPath();
      ctx.moveTo(x, plane.paddingTop);
      ctx.lineTo(x, plane.paddingTop + plane.plotHeight);
      ctx.stroke();

      ctx.fillStyle = '#d8704f';
      ctx.beginPath();
      ctx.arc(x, yToPx(conditionalInfo.conditionalMu), 5, 0, TAU);
      ctx.fill();
    } else {
      const y = yToPx(conditionalInfo.observedValue);
      ctx.beginPath();
      ctx.moveTo(plane.paddingLeft, y);
      ctx.lineTo(plane.paddingLeft + plane.plotWidth, y);
      ctx.stroke();

      ctx.fillStyle = '#d8704f';
      ctx.beginPath();
      ctx.arc(xToPx(conditionalInfo.conditionalMu), y, 5, 0, TAU);
      ctx.fill();
    }
  }
}

function getConditionalInfo(state) {
  const offset = state.conditionOffset;
  if (state.conditionAxis === 'x') {
    const observedValue = state.muX + offset * state.sigmaX;
    const conditionalMu = state.muY + state.rho * (state.sigmaY / state.sigmaX) * (observedValue - state.muX);
    const conditionalSigma = Math.sqrt(1 - state.rho ** 2) * state.sigmaY;
    return {
      observedValue,
      conditionalMu,
      conditionalSigma,
      observedLabel: `X = ${observedValue.toFixed(2)}`,
      title: `Y | X = ${observedValue.toFixed(2)}`,
      formula: `N(${conditionalMu.toFixed(2)}, ${conditionalSigma.toFixed(2)}²)`,
      otherVariable: 'Y',
      copy: Math.abs(state.rho) < 0.05
        ? 'With almost zero correlation, conditioning barely changes the y distribution.'
        : 'Because x and y move together, pinning x down makes the y distribution narrower and shifts its mean along the tilt of the cloud.',
    };
  }

  const observedValue = state.muY + offset * state.sigmaY;
  const conditionalMu = state.muX + state.rho * (state.sigmaX / state.sigmaY) * (observedValue - state.muY);
  const conditionalSigma = Math.sqrt(1 - state.rho ** 2) * state.sigmaX;
  return {
    observedValue,
    conditionalMu,
    conditionalSigma,
    observedLabel: `Y = ${observedValue.toFixed(2)}`,
    title: `X | Y = ${observedValue.toFixed(2)}`,
    formula: `N(${conditionalMu.toFixed(2)}, ${conditionalSigma.toFixed(2)}²)`,
    otherVariable: 'X',
    copy: Math.abs(state.rho) < 0.05
      ? 'With almost zero correlation, conditioning barely changes the x distribution.'
      : 'A slice across the tilted cloud leaves a narrower Gaussian on the other axis, centered where the tilt predicts it should be.',
  };
}

    elements.progressFill.style.width = `${Math.min(100, Math.max(0, progress * 100))}%`;
  };

  window.addEventListener('scroll', onScroll, { passive: true });
  onScroll();
}
